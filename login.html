//login.js import { auth } from "./firebase.js"; import { signInWithEmailAndPassword } from "firebase/auth"; function login() { const email = document.getElementById("email").value.trim(); const password = document.getElementById("password").value.trim(); if (!email || !password) { alert("Enter email and password"); return; } signInWithEmailAndPassword(auth, email, password) .then((userCredential) => { alert("Login Successful ✅"); console.log("User:", userCredential.user.email); }) .catch((error) => { alert(error.code); console.log(error); }); } window.login = login;
